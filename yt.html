<!DOCTYPE html> <html lang="en"> <head> <meta charset="UTF-8" /> <meta name="viewport" content="width=device-width, initial-scale=1.0"/> <title>Inventory</title> <script src="/socket.io/socket.io.js"></script> </head> <body> <h1>Inventory List</h1> <ul id="product-list"></ul> <h2>Add New Product</h2> <form id="product-form"> Name: <input type="text" id="name" required /><br/> Category: <input type="text" id="category" required /><br/> Price: <input type="number" id="price" required /><br/> Quantity: <input type="number" id="quantity" required /><br/> <button type="submit">Add Product</button> </form> <script> const socket = io(); const productList = document.getElementById('product-list'); // Function to fetch and display products async function loadProducts() { const res = await fetch('/api/products'); const products = await res.json(); productList.innerHTML = ''; products.forEach(p => { const li = document.createElement('li'); li.textContent = `${p.name} (${p.category}) - $${p.price} - Qty: ${p.quantity}`; productList.appendChild(li); }); } // Listen for real-time updates socket.on('stockUpdated', (product) => { loadProducts(); // Refresh list }); // Submit form to add product document.getElementById('product-form').addEventListener('submit', async (e) => { e.preventDefault(); const newProduct = { name: document.getElementById('name').value, category: document.getElementById('category').value, price: parseFloat(document.getElementById('price').value), quantity: parseInt(document.getElementById('quantity').value), }; await fetch('/api/products', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(newProduct), }); document.getElementById('product-form').reset(); }); // Initial load loadProducts(); </script> </body> </html>
